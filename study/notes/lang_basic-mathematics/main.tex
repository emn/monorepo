Commutativity. \text{If $a, b$ are integers, then} $a + b = b + a$ \\
Associativity. \text{If $a, b, c$ are integers, then} $(a + b) + c = a + (b + c)$ \\

N3. \textit{If $a + b = 0$ then $b = -a$ and $a = -b$}
\begin{proof}
 \begin{align*}
	&a + b = 0 \\
	&a + b + (-b) = 0 + (-b) \\
	&a + b - b = 0 - b \\
	&a = -b &
 \end{align*}
\end{proof}

\leavevmode \\
N4. $a = -(-a)$ \\
\begin{proof}
Let $b = -a$
\begin{align*}
&a + b = 0	&\text{N2}\\
&a = -b		&\text{N3}\\
&a = -(-a)
\end{align*}
\end{proof}

\leavevmode \\
N5. $-(a+b) = -a - b$ \\
\begin{proof}
Let $x = a+b$ and $y = -a - b$
\begin{align*}
&x + y = 0 \\
&a + b - a - b = 0 \\
&a - a + b - b = 0 &\text{commutativity} \\
&0 + 0 = 0
\end{align*}
therefore
\begin{align*}
&-x = y &\text{N3} \\
&-(a+b) = -a - b
\end{align*}
\end{proof}

\textbf{Exercises}\\
Justify each step, using commutativity and associativity in proving the following identities. \\
\begin{enumerate}
\item $(a + b) + (c + d) = (a + d) + (b + c)$
	\begin{align*}
	 &= a + b + c + d = a + d + b + c	&&\text{associativity} \\
	 &= a + b + c + d = a + b + c + d	&&\text{commutativity} \\
	\end{align*}
\item $(a + b) + (c + d) = (a + c) + (b + d)$
	\begin{align*}
	 &= a + b + c + d = a + c + b + d	&&\text{associativity} \\
	 &= a + b + c + d = a + b + c + d	&&\text{commutativity} \\
	\end{align*}
\item $(a - b) + (c - d) = (a + c) + ( -b - d)$
	\begin{align*}
	 &= a - b + c - d = a + c - b - d	&&\text{associativity} \\
	 &= a - b + c - d = a - b + c - d	&&\text{commutativity} \\
	\end{align*}
\item $(a - b) + (c - d) = (a + c) - (b + d)$
	\begin{align*}
	 &= a - b + c - d = a + c - (b + d)	&&\text{associativity} \\
	 &= a - b + c - d = a + c - b - d	&&\text{N5} \\
	 &= a - b + c - d = a - b + c - d	&&\text{commutativity} \\
	\end{align*}
\item $(a - b) + (c - d) = (a - d) + (c - b)$
	\begin{align*}
	 &= a - b + c - d = a - d + c - b	&&\text{associativity} \\
	 &= a - b + c - d = a - b + c - d	&&\text{commutativity} \\
	\end{align*}
\item $(a - b) + (c - d) = -(b + d) + (a + c)$
	\begin{align*}
	 &= a - b + c - d = -b - d + (a + c)	&&\text{N5} \\
	 &= a - b + c - d = -b - d + a + c	&&\text{associativity} \\
	 &= a - b + c - d = a - b + c - d	&&\text{commutativity} \\
	\end{align*}
\item $(a - b) + (c - d) = -(b + d) - (-a - c)$
	\begin{align*}
	 &= a - b + c - d = -b - d - (-(a+c))		&&\text{N5} \\
	 &= a - b + c - d = -b - d + (a+c)		&&\text{N4} \\
	 &= a - b + c - d = -b - d + a + c		&&\text{associativity} \\
	 &= a - b + c - d = a - b + c - d		&&\text{commutativity} \\
	\end{align*}
\item $((x + y) + z) + w = (x + z) + (y + w)$
	\begin{align*}
	 &= x + y + z + w = x + z + y + w		&&\text{associativity} \\
	 &= x + y + z + w = x + y + z + w		&&\text{commutativity} \\
	\end{align*}
\item $(x - y) - (z - w) = (x + w) - y - z$
	\begin{align*}
	 &= x - y - (z + (-w)) = x + w - y - z	&&\text{associativity} \\
	 &= x - y - z - (-w) = x + w - y - z		&&\text{N5} \\
	 &= x - y - z + w = x + w - y - z		&&\text{N4} \\
	 &= x - y - z + w = x - y - z + w		&&\text{commutativity}
	\end{align*}
\item $(x - y) - (z - w) = (x - z) + (w - y)$
	\begin{align*}
	 &= x - y - (z + (-w)) = x - z + w - y	&&\text{associativity} \\
	 &= x - y - z - (-w) = x - z + w - y		&&\text{N5} \\
	 &= x - y - z + w = x - z + w - y		&&\text{N4} \\
	 &= x - y - z + w = x - y - z + w		&&\text{commutativity}
	\end{align*}
\item Show that $-(a+b+c) = -a + (-b) + (-c)$
\begin{proof}
Let $x = a+b+c$ and $y = -a + (-b) + (-c)$
\begin{align*}
&x + y = 0 \\
&a + b + c -a + (-b) + (-c) = 0 \\
&a + (-a) + b + (-b) + c + (-c) = 0 &&\text{commutativity} \\
&0 + 0 + 0 = 0
\end{align*}
therefore
\begin{align*}
&-x = y &\text{N3} \\
&-(a+b+c) = -a + (-b) + (-c)
\end{align*}
\end{proof}
\item Show that $-(a-b-c) = -a + b + c$
\begin{proof}
Let $x = a-b-c$ and $y = -a + b + c$
\begin{align*}
&x + y = 0 \\
&a-b-c -a + b + c = 0 \\
&a-a + b -b + c -c = 0 &&\text{commutativity} \\
&0 + 0 + 0 = 0
\end{align*}
therefore
\begin{align*}
&-x = y &\text{N3} \\
&-(a-b-c) = -a + b + c
\end{align*}
\end{proof}
\item Show that $-(a-b)=b-a$
\begin{proof}
\begin{align*}
&-(a-b)\\
&-((-b)+a) &&\text{commutativity} \\
&-(-b)-a &&\text{N5} \\
&b-a &&\text{N4}
\end{align*}
\end{proof}
\end{enumerate}

\subsection{Rules for Multiplication}
Commutativity. $ab = ba$ \\
Associativity. $(ab)c = a(bc)$ \\
N6. $1a=a$ and $0a=0$\\
Distributivity. $a(b+c) = ab+ac$\\
N7. $(-1)a = -a$
\begin{proof}
 \begin{align*}
	&(-1)a = -a\\
	&(-1)a + a = -a + a\\
	&(-1)a + 1a = 0\\
	&(-1 + 1)a = 0 &&\text{distributivity}\\
	&0a = 0
 \end{align*}
\end{proof}
\leavevmode \\
N8. $-(ab) = (-a)b$
\begin{proof}
 \begin{align*}
	&ab + (-a)b = 0\\
	&(a+-a)b = 0 &&\text{distributivity}\\
	&0b = 0\\
 \end{align*}
\end{proof}
\leavevmode \\
N9. $-(ab) = a(-b)$
\begin{proof}
 \begin{align*}
	&ab + a(-b) = 0\\
	&a(b-b) = 0 &&\text{distributivity}\\
	&a0 = 0\\
 \end{align*}
\end{proof}

N10. $(-a)(-b) = ab$

 \begin{align*}

	&-a(-b)\\
	&-(a(-b))&&\text{N9}\\
	&-(-ab))&&\text{N9}\\

 \end{align*}
\end{proof}
